<script>
verificarPermissao("config");

function carregarDashboard() {
    const producao = listar("producao");
    const produtos = listar("produtos");
    const insumos = listar("insumos");
    const logs = listar("logs");

    const hoje = new Date().toLocaleDateString();
    const prodHoje = producao.filter(p => p.data.startsWith(hoje));
    const totalProdHoje = prodHoje.reduce((s, p) => s + Number(p.quantidade), 0);
    document.getElementById("prodDia").innerText = totalProdHoje;

    const custoHoje = prodHoje.reduce((s, p) => s + Number(p.custo), 0);
    document.getElementById("custoDia").innerText = "R$ " + custoHoje.toFixed(2);

    const criticos = produtos.filter(p => {
        const total = producao.filter(x => x.produtoId == p.id).reduce((s, x) => s + Number(x.quantidade), 0);
        return total < p.estoqueMinimo;
    });
    document.getElementById("estoqueCritico").innerText = criticos.length;

    const hojeData = new Date();
    const vencendo = insumos.filter(i => {
        const validade = new Date(i.validade);
        const diff = (validade - hojeData) / (1000 * 60 * 60 * 24);
        return diff >= 0 && diff <= 5;
    });
    document.getElementById("insumosVencendo").innerText = vencendo.length;

    const tabela = document.getElementById("listaLogs");
    tabela.innerHTML = "";
    logs.slice(-10).reverse().forEach(l => {
        tabela.innerHTML += `<tr><td>${l.usuario}</td><td>${l.acao}</td><td>${l.detalhes}</td><td>${l.data}</td></tr>`;
    });
}

carregarDashboard();
</script>
