<script>
const id = localStorage.getItem("editUsuario");
const lista = listar("usuarios");
const user = lista.find(x => x.id == id);

nome.value = user.nome;
email.value = user.email;
senha.value = user.senha;
permissao.value = user.permissao;

function salvar() {
    user.nome = nome.value;
    user.email = email.value;
    user.senha = senha.value;
    user.permissao = permissao.value;

    atualizar("usuarios", user.id, user);
    registrarLog("Editou usuário", user.email);
    location.href = "admin.html";
}
</script>
