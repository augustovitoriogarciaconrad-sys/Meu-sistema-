<script>
function carregarUsuarios() {
    const usuarios = listar("usuarios");
    const tabela = document.getElementById("listaUsuarios");
    tabela.innerHTML = "";

    usuarios.forEach(u => {
        tabela.innerHTML += `
        <tr>
            <td>${u.nome}</td>
            <td>${u.email}</td>
            <td>${u.permissao}</td>
            <td>
                <button class="btn-secondary" onclick="editar(${u.id})">Editar</button>
                <button class="btn-danger" onclick="excluirUsuario(${u.id})">Excluir</button>
            </td>
        </tr>`;
    });
}

function editar(id) {
    localStorage.setItem("editUsuario", id);
    location.href = "usuario_edit.html";
}

function excluirUsuario(id) {
    remover("usuarios", id);
    registrarLog("Excluiu usuário", "ID " + id);
    carregarUsuarios();
}

function carregarLogs() {
    const logs = listar("logs");
    const tabela = document.getElementById("listaLogs");
    tabela.innerHTML = "";

    logs.slice(-30).reverse().forEach(l => {
        tabela.innerHTML += `
        <tr>
            <td>${l.usuario}</td>
            <td>${l.acao}</td>
            <td>${l.detalhes}</td>
            <td>${l.data}</td>
        </tr>`;
    });
}

carregarUsuarios();
carregarLogs();
</script>
