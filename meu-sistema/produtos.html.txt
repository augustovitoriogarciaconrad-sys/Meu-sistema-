<script>
function carregarProdutos() {
    const produtos = listar("produtos");
    const tabela = document.getElementById("listaProdutos");
    tabela.innerHTML = "";

    produtos.forEach(p => {
        tabela.innerHTML += `
        <tr>
            <td>${p.nome}</td>
            <td>${p.categoria}</td>
            <td>R$ ${Number(p.preco).toFixed(2)}</td>
            <td>${p.estoqueMinimo}</td>
            <td>
                <button class="btn-secondary" onclick="editar(${p.id})">Editar</button>
                <button class="btn-danger" onclick="excluirProduto(${p.id})">Excluir</button>
            </td>
        </tr>`;
    });
}

function editar(id) {
    localStorage.setItem("editProduto", id);
    location.href = "produto_edit.html";
}

function excluirProduto(id) {
    remover("produtos", id);
    registrarLog("Excluiu produto", "ID " + id);
    carregarProdutos();
}

carregarProdutos();
</script>
