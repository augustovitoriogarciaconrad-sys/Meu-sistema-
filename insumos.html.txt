<script>
function carregarInsumos() {
    const insumos = listar("insumos");
    const tabela = document.getElementById("listaInsumos");
    tabela.innerHTML = "";

    insumos.forEach(i => {
        tabela.innerHTML += `
        <tr>
            <td>${i.nome}</td>
            <td>${i.categoria}</td>
            <td>${i.quantidade}</td>
            <td>${i.validade}</td>
            <td>
                <button class="btn-secondary" onclick="editar(${i.id})">Editar</button>
                <button class="btn-danger" onclick="excluirInsumo(${i.id})">Excluir</button>
            </td>
        </tr>`;
    });
}

function editar(id) {
    localStorage.setItem("editInsumo", id);
    location.href = "insumo_edit.html";
}

function excluirInsumo(id) {
    remover("insumos", id);
    registrarLog("Excluiu insumo", "ID " + id);
    carregarInsumos();
}

carregarInsumos();
</script>
