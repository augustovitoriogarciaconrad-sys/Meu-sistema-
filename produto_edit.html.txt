<input id="preco" type="number">

<label>Estoque Mínimo</label>
<input id="estoqueMinimo" type="number">

<button class="btn-primary" onclick="salvar()">Salvar</button>
<button class="btn-secondary" onclick="history.back()">Cancelar</button>

<script src="js/app.js"></script>
<script>
const id = localStorage.getItem("editProduto");
const lista = listar("produtos");
const prod = lista.find(x => x.id == id);

nome.value = prod.nome;
categoria.value = prod.categoria;
preco.value = prod.preco;
estoqueMinimo.value = prod.estoqueMinimo;

function salvar() {
    prod.nome = nome.value;
    prod.categoria = categoria.value;
    prod.preco = preco.value;
    prod.estoqueMinimo = estoqueMinimo.value;

    atualizar("produtos", prod.id, prod);
    registrarLog("Editou produto", prod.nome);
    location.href = "produtos.html";
}
</script>
