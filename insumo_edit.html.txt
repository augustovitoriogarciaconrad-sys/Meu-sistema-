<script src="js/app.js"></script>
<script>
const id = localStorage.getItem("editInsumo");
const lista = listar("insumos");
const ins = lista.find(x => x.id == id);

nome.value = ins.nome;
categoria.value = ins.categoria;
quantidade.value = ins.quantidade;
validade.value = ins.validade;

function salvar() {
    ins.nome = nome.value;
    ins.categoria = categoria.value;
    ins.quantidade = quantidade.value;
    ins.validade = validade.value;

    atualizar("insumos", ins.id, ins);
    registrarLog("Editou insumo", ins.nome);
    location.href = "insumos.html";
}
</script>
